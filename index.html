main.js:

document.body.appendChild(imgElement);

webpack.config.js:

                        
module.exports = {
  entry: './main.js',
  output: {
    filename: 'bundle.js'       
  },
  module: {
    loaders: [
      {test: /\.js$/, loader: "6to5"}, // transpiles es6
      {test: /\.scss$/, loader: "style!css!sass"},
      {test: /\.png$/, loader: "url-loader?limit=10000"}
    ]
  }
};

        });
